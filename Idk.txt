local OrionLib = {
    Elements = {},
    ThemeObjects = {},
    Connections = {},
    Flags = {},
    Themes = {
        Default = {
            Main = Color3.fromRGB(25, 25, 25),
            Second = Color3.fromRGB(32, 32, 32),
            Stroke = Color3.fromRGB(60, 60, 60),
            Divider = Color3.fromRGB(60, 60, 60),
            Text = Color3.fromRGB(240, 240, 240),
            TextDark = Color3.fromRGB(150, 150, 150)
        }
    },
    SelectedTheme = "Default",
    Folder = nil,
    SaveCfg = false
}

function OrionLib:AddTheme(themeName, themeData)
    self.Themes[themeName] = themeData
end

function OrionLib:SetTheme(themeName)
    local colors = self.Themes[themeName]
    if not colors then
        warn("テーマが存在しません: "..tostring(themeName))
        return
    end

    self.SelectedTheme = themeName

    local propertyMap = {
        Main = "BackgroundColor3",
        Second = "BackgroundColor3",
        Stroke = "Color",
        Divider = "BackgroundColor3",
        Text = "TextColor3",
        TextDark = "TextColor3"
    }

    for _, obj in pairs(self.ThemeObjects) do
        if obj.Instance and obj.Type then
            local prop = propertyMap[obj.Type]
            if prop and colors[obj.Type] then
                pcall(function()
                    obj.Instance[prop] = colors[obj.Type]
                end)
            end
        end
    end
end

local function GetIcon(IconName)
	if not IconName then

		return nil

	IconName = string.lower(IconName)
	local assetId = LucideIcons[IconName]

local function AddConnection(Signal, Function)
	if (not OrionLib:IsRunning()) then
		return
	end
	local SignalConnect = Signal:Connect(Function)
	table.insert(OrionLib.Connections, SignalConnect)
	return SignalConnect
end

	while (OrionLib:IsRunning()) do
		wait()
	end

	for _, Connection in next, OrionLib.Connections do
		Connection:Disconnect()
	end
end)

local function MakeDraggable(DragPoint, Main)
    pcall(function()
        local Dragging, DragInput, MousePos, FramePos = false
        local TargetPos = Main.Position

        AddConnection(RunService.RenderStepped, function()
            Main.Position = Main.Position:Lerp(TargetPos, 0.25)
        end)

        AddConnection(DragPoint.InputBegan, function(Input)
            if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                Dragging = true
                MousePos = Input.Position
                FramePos = Main.Position

                Input.Changed:Connect(function()
                    if Input.UserInputState == Enum.UserInputState.End then
                        Dragging = false
                    end
                end)
            end
        end)

        AddConnection(DragPoint.InputChanged, function(Input)
            if Input.UserInputType == Enum.UserInputType.MouseMovement or Input.UserInputType == Enum.UserInputType.Touch then
                DragInput = Input
            end
        end)

        AddConnection(UserInputService.InputChanged, function(Input)
            if Input == DragInput and Dragging then
                local Delta = Input.Position - MousePos
                TargetPos = UDim2.new(FramePos.X.Scale,FramePos.X.Offset + Delta.X, FramePos.Y.Scale, FramePos.Y.Offset + Delta.Y)
            end
        end)
    end)
end

	return Object
end

local function Round(Number, Factor)
	local Result = math.floor(Number/Factor + (math.sign(Number) * 0.5)) * Factor
	if Result < 0 then Result = Result + Factor end
	return Result
end

local function ReturnProperty(Object)
	if Object:IsA("Frame") or Object:IsA("TextButton") then
		return "BackgroundColor3"

		return "ScrollBarImageColor3"

	if Object:IsA("UIStroke") then
		return "Color"
	end 
	if Object:IsA("TextLabel") or Object:IsA("TextBox") then
		return "TextColor3"
	end   
	if Object:IsA("ImageLabel") or Object:IsA("ImageButton") then
		return "ImageColor3"
	end   
end

	end    
end

local function PackColor(Color)
	return {R = Color.R * 255, G = Color.G * 255, B = Color.B * 255}
end    

local function UnpackColor(Color)
	return Color3.fromRGB(Color.R, Color.G, Color.B)
end

local function LoadCfg(Config)
	local Data = HttpService:JSONDecode(Config)
	table.foreach(Data, function(a,b)
		if OrionLib.Flags[a] then
			spawn(function() 
				if OrionLib.Flags[a].Type == "Colorpicker" then
					OrionLib.Flags[a]:Set(UnpackColor(b))
				else
					OrionLib.Flags[a]:Set(b)
				end    
			end)
		else
			warn("Orion Library Config Loader - Could not find ", a ,b)
		end
	end)
end

local function SaveCfg(Name)
	local Data = {}
	for i,v in pairs(OrionLib.Flags) do
		if v.Save then
			if v.Type == "Colorpicker" then
				Data[i] = PackColor(v.Value)
			else
				Data[i] = v.Value
			end
		end	
	end
	writefile(OrionLib.Folder .. "/" .. Name .. ".txt", tostring(HttpService:JSONEncode(Data)))
end

local notificationCount = 0

function OrionLib:MakeNotification(NotificationConfig)
    task.spawn(function()
        NotificationConfig = {
            Name = NotificationConfig.Name or "Notification",
            Content = NotificationConfig.Content or "Test",
            Image = NotificationConfig.Image or "rbxassetid://4384403532",
            Time = NotificationConfig.Time or 15,
            BackgroundColor = NotificationConfig.BackgroundColor or Color3.fromRGB(25, 25, 30),
            AccentColor = NotificationConfig.AccentColor or Color3.fromRGB(0, 200, 255)
        }

        notificationCount = notificationCount + 1
        local currentLayoutOrder = notificationCount

        local NotificationParent = SetProps(MakeElement("TFrame"), {
            Size = UDim2.new(1, 0, 0, 0),
            AutomaticSize = Enum.AutomaticSize.Y,
            Parent = NotificationHolder,
            ClipsDescendants = false,
            LayoutOrder = currentLayoutOrder,
            BackgroundTransparency = 1
        })

        local NotificationFrame = SetChildren(
            SetProps(MakeElement("RoundFrame", NotificationConfig.BackgroundColor, 0, 12), {
                Parent = NotificationParent,
                Size = UDim2.new(1, 0, 0, 0),
                Position = UDim2.new(1.2, 0, 0, 0),
                BackgroundTransparency = 1,
                AutomaticSize = Enum.AutomaticSize.Y
            }),
            {
                MakeElement("Padding", 15, 15, 15, 15),
                
                SetProps(MakeElement("Image", NotificationConfig.Image), {
                    Size = UDim2.new(0, 24, 0, 24),
                    ImageColor3 = NotificationConfig.AccentColor,
                    ImageTransparency = 1,
                    Name = "Icon"
                }),
                
                SetProps(MakeElement("Label", NotificationConfig.Name, 16), {
                    Size = UDim2.new(1, -65, 0, 24),
                    Position = UDim2.new(0, 34, 0, 0),
                    Font = Enum.Font.GothamBold,
                    Name = "Title",
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextTransparency = 1
                }),
                
                SetProps(MakeElement("Label", NotificationConfig.Content, 14), {
                    Size = UDim2.new(1, -10, 0, 0),
                    Position = UDim2.new(0, 0, 0, 32),
                    Font = Enum.Font.Gotham,
                    Name = "Content",
                    AutomaticSize = Enum.AutomaticSize.Y,
                    TextColor3 = Color3.fromRGB(200, 200, 200),
                    TextWrapped = true,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextTransparency = 1
                })
            }
        )

        local NotificationStroke = SetProps(MakeElement("Stroke", NotificationConfig.AccentColor, 1.5), {
            Transparency = 1,
            Parent = NotificationFrame
        })

        local TimerBarTrack = SetProps(MakeElement("RoundFrame", Color3.fromRGB(40, 40, 50), 0.7, 6), {
            Parent = NotificationFrame,
            Size = UDim2.new(1, 0, 0, 4),
            AnchorPoint = Vector2.new(0, 1)
        })

        local TimerBar = SetChildren(
            SetProps(MakeElement("RoundFrame", NotificationConfig.AccentColor, 1, 6), {
                Parent = TimerBarTrack,
                Size = UDim2.new(1, 0, 1, 0),
                Position = UDim2.new(0, 0, 0, 0)
            }),
            {
                MakeElement("UIGradient", {
                    Color = ColorSequence.new({
                        ColorSequenceKeypoint.new(0, NotificationConfig.AccentColor:Lerp(Color3.fromRGB(255, 255, 255), 0.3)),
                        ColorSequenceKeypoint.new(0.5, NotificationConfig.AccentColor),
                        ColorSequenceKeypoint.new(1, NotificationConfig.AccentColor:Lerp(Color3.fromRGB(0, 0, 0), 0.2))
                    }),
                    Rotation = 90
                }),
                SetProps(MakeElement("Stroke", NotificationConfig.AccentColor, 1.2), {
                    Transparency = 0.5
                })
            }
        )

        local ParticleHolder = SetProps(MakeElement("TFrame"), {
            Size = UDim2.new(1, 0, 1, 0),
            Parent = NotificationFrame,
            ZIndex = 0
        })

        local CloseBtnNotify = SetChildren(
            SetProps(MakeElement("Button"), {
                Size = UDim2.new(0, 24, 0, 24),
                Position = UDim2.new(1, -30, 0, 0),
                BackgroundColor3 = Color3.fromRGB(40, 40, 45),
                BackgroundTransparency = 1,
                Text = "",
                AutoButtonColor = false
            }),
            {
                MakeElement("Corner", 12),
                SetProps(MakeElement("Stroke", Color3.fromRGB(80, 80, 90), 1.2), {
                    Transparency = 1
                }),
                SetProps(MakeElement("Image", "rbxassetid://7072725342"), {
                    Size = UDim2.new(0.6, 0, 0.6, 0),
                    Position = UDim2.new(0.2, 0, 0.2, 0),
                    ImageColor3 = Color3.fromRGB(200, 200, 210),
                    ImageTransparency = 1
                })
            }
        )
        CloseBtnNotify.Parent = NotificationFrame

        local CloseIcon = CloseBtnNotify:FindFirstChildOfClass("Image")
        local CloseStroke = CloseBtnNotify:FindFirstChildOfClass("Stroke")

        CloseBtnNotify.MouseEnter:Connect(function()
            TweenService:Create(
                CloseBtnNotify,
                TweenInfo.new(0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
                {BackgroundColor3 = Color3.fromRGB(255, 60, 60), BackgroundTransparency = 0}
            ):Play()
            TweenService:Create(
                CloseStroke,
                TweenInfo.new(0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
                {Color = Color3.fromRGB(255, 100, 100), Transparency = 0.3}
            ):Play()
            TweenService:Create(
                CloseIcon,
                TweenInfo.new(0.2, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
                {ImageTransparency = 0, Rotation = 90, ImageColor3 = Color3.fromRGB(255, 255, 255)}
            ):Play()
        end)

        CloseBtnNotify.MouseLeave:Connect(function()
            TweenService:Create(
                CloseBtnNotify,
                TweenInfo.new(0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
                {BackgroundColor3 = Color3.fromRGB(40, 40, 45), BackgroundTransparency = 0.5}
            ):Play()
            TweenService:Create(
                CloseStroke,
                TweenInfo.new(0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
                {Color = Color3.fromRGB(80, 80, 90), Transparency = 0.6}
            ):Play()
            TweenService:Create(
                CloseIcon,
                TweenInfo.new(0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
                {ImageTransparency = 0.3, Rotation = 0, ImageColor3 = Color3.fromRGB(200, 200, 210)}
            ):Play()
        end)

        local isRemoving = false
        local function RemoveNotification()
            if isRemoving then return end
            isRemoving = true
            
            local fadeOutTween = TweenInfo.new(0.4, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
            
            for i = 1, 12 do
                task.spawn(function()
                    local particle = SetProps(MakeElement("RoundFrame", NotificationConfig.AccentColor, 0, 4), {
                        Size = UDim2.new(0, 4, 0, 4),
                        Position = UDim2.new(0.5, 0, 0.5, 0),
                        Parent = ParticleHolder,
                        ZIndex = 10
                    })
                    
                    local angle = (i / 12) * math.pi * 2
                    local distance = math.random(40, 80)
                    local targetX = math.cos(angle) * distance
                    local targetY = math.sin(angle) * distance
                    
                    TweenService:Create(
                        particle,
                        TweenInfo.new(0.6, Enum.EasingStyle.Quart, Enum.EasingDirection.Out),
                        {
                            Position = UDim2.new(0.5, targetX, 0.5, targetY),
                            BackgroundTransparency = 1,
                            Size = UDim2.new(0, 0, 0, 0)
                        }
                    ):Play()
                end)
            end
            
            TweenService:Create(NotificationFrame, fadeOutTween, {
                BackgroundTransparency = 1,
                Position = UDim2.new(1.2, 0, 0, 0)
            }):Play()
            
            TweenService:Create(NotificationStroke, fadeOutTween, {
                Transparency = 1
            }):Play()
            
            TweenService:Create(NotificationFrame.Title, fadeOutTween, {TextTransparency = 1}):Play()
            TweenService:Create(NotificationFrame.Content, fadeOutTween, {TextTransparency = 1}):Play()
            
            TweenService:Create(NotificationFrame.Icon, fadeOutTween, {ImageTransparency = 1}):Play()
            
            TweenService:Create(TimerBarTrack, fadeOutTween, {BackgroundTransparency = 1}):Play()
            TweenService:Create(TimerBar, fadeOutTween, {BackgroundTransparency = 1}):Play()
            if TimerBar:FindFirstChildOfClass("Stroke") then
                TweenService:Create(TimerBar:FindFirstChildOfClass("Stroke"), fadeOutTween, {Transparency = 1}):Play()
            end
            
            TweenService:Create(CloseBtnNotify, fadeOutTween, {BackgroundTransparency = 1}):Play()
            TweenService:Create(CloseIcon, fadeOutTween, {ImageTransparency = 1}):Play()
            TweenService:Create(CloseStroke, fadeOutTween, {Transparency = 1}):Play()
            
            task.wait(0.4)
            
            if NotificationParent and NotificationParent.Parent then
                NotificationParent:Destroy()
            end
        end

        CloseBtnNotify.MouseButton1Click:Connect(RemoveNotification)

        task.wait()
        
        local fadeInTween = TweenInfo.new(0.6, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
        local quickFadeIn = TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
        
        TweenService:Create(NotificationFrame, fadeInTween, {
            Position = UDim2.new(0, 0, 0, 0),
            BackgroundTransparency = 0
        }):Play()
        
        TweenService:Create(NotificationStroke, quickFadeIn, {
            Transparency = 0.5
        }):Play()
        
        task.spawn(function()
            task.wait(0.3)
            while NotificationStroke and NotificationStroke.Parent do
                TweenService:Create(
                    NotificationStroke,
                    TweenInfo.new(1.5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true),
                    {Transparency = 0.2}
                ):Play()
                break
            end
        end)
        
        task.wait(0.1)
        TweenService:Create(NotificationFrame.Icon, quickFadeIn, {ImageTransparency = 0}):Play()
        
        task.wait(0.05)
        TweenService:Create(NotificationFrame.Title, quickFadeIn, {TextTransparency = 0}):Play()
        
        task.wait(0.05)
        TweenService:Create(NotificationFrame.Content, quickFadeIn, {TextTransparency = 0}):Play()
        
        task.wait(0.05)
        TweenService:Create(CloseBtnNotify, quickFadeIn, {BackgroundTransparency = 0.5}):Play()
        TweenService:Create(CloseStroke, quickFadeIn, {Transparency = 0.6}):Play()
        TweenService:Create(CloseIcon, quickFadeIn, {ImageTransparency = 0.3}):Play()

        RunService.Heartbeat:Wait()
        local contentBottom = NotificationFrame.Content.AbsolutePosition.Y + NotificationFrame.Content.AbsoluteSize.Y
        local frameTop = NotificationFrame.AbsolutePosition.Y
        local relativeY = contentBottom - frameTop
        TimerBarTrack.Position = UDim2.new(0, 0, 0, relativeY + 10)

        TweenService:Create(TimerBarTrack, quickFadeIn, {BackgroundTransparency = 0.7}):Play()
        TweenService:Create(TimerBar, quickFadeIn, {BackgroundTransparency = 0}):Play()
        if TimerBar:FindFirstChildOfClass("Stroke") then
            TweenService:Create(TimerBar:FindFirstChildOfClass("Stroke"), quickFadeIn, {Transparency = 0.5}):Play()
        end

        task.wait(0.3)
        TweenService:Create(
            TimerBar,
            TweenInfo.new(NotificationConfig.Time, Enum.EasingStyle.Linear),
            {Size = UDim2.new(0, 0, 1, 0)}
        ):Play()

        task.delay(NotificationConfig.Time, RemoveNotification)
    end)
end

-- ===== DIALOG SYSTEM =====
function OrionLib:MakeDialog(DialogConfig)
    task.spawn(function()
        DialogConfig = {
            Title = DialogConfig.Title or "Confirmation",
            Content = DialogConfig.Content or "Are you sure?",
            Image = DialogConfig.Image or "rbxassetid://4384403532",
            Buttons = DialogConfig.Buttons or {
                {Name = "Confirm", Callback = function() end},
                {Name = "Cancel", Callback = function() end}
            },
            BackgroundColor = DialogConfig.BackgroundColor or Color3.fromRGB(25, 25, 30),
            AccentColor = DialogConfig.AccentColor or Color3.fromRGB(0, 200, 255)
        }

        dialogCount = dialogCount + 1

        -- オーバーレイ
        local Overlay = SetProps(MakeElement("Frame", Color3.fromRGB(0, 0, 0)), {
            Size = UDim2.new(1, 0, 1, 0),
            Position = UDim2.new(0, 0, 0, 0),
            BackgroundTransparency = 1,
            ZIndex = 1000,
            Parent = ScreenGui
        })

        -- ダイアログフレーム
        local DialogFrame = SetChildren(
            SetProps(MakeElement("RoundFrame", DialogConfig.BackgroundColor, 0, 16), {
                Parent = Overlay,
                Size = UDim2.new(0, 400, 0, 0),
                Position = UDim2.new(0.5, -200, 0.5, -150),
                BackgroundTransparency = 1,
                AutomaticSize = Enum.AutomaticSize.Y,
                AnchorPoint = Vector2.new(0, 0.5)
            }),
            {
                MakeElement("Padding", 20, 20, 20, 20),
                
                SetProps(MakeElement("Image", DialogConfig.Image), {
                    Size = UDim2.new(0, 32, 0, 32),
                    Position = UDim2.new(0.5, -16, 0, 0),
                    ImageColor3 = DialogConfig.AccentColor,
                    ImageTransparency = 1,
                    Name = "Icon"
                }),
                
                SetProps(MakeElement("Label", DialogConfig.Title, 20), {
                    Size = UDim2.new(1, 0, 0, 30),
                    Position = UDim2.new(0, 0, 0, 45),
                    Font = Enum.Font.GothamBold,
                    Name = "Title",
                    TextTransparency = 1
                }),
                
                SetProps(MakeElement("Label", DialogConfig.Content, 14), {
                    Size = UDim2.new(1, 0, 0, 0),
                    Position = UDim2.new(0, 0, 0, 85),
                    Font = Enum.Font.Gotham,
                    Name = "Content",
                    AutomaticSize = Enum.AutomaticSize.Y,
                    TextColor3 = Color3.fromRGB(200, 200, 200),
                    TextWrapped = true,
                    TextTransparency = 1
                })
            }
        )

        local DialogStroke = SetProps(MakeElement("Stroke", DialogConfig.AccentColor, 2), {
            Transparency = 1,
            Parent = DialogFrame
        })

        -- ボタンコンテナ
        local ButtonContainer = SetProps(MakeElement("TFrame"), {
            Size = UDim2.new(1, 0, 0, 45),
            AutomaticSize = Enum.AutomaticSize.Y,
            BackgroundTransparency = 1,
            Parent = DialogFrame
        })

        SetProps(MakeElement("List", 0, 10), {
            Parent = ButtonContainer,
            FillDirection = Enum.FillDirection.Horizontal,
            HorizontalAlignment = Enum.HorizontalAlignment.Center
        })

        RunService.Heartbeat:Wait()
        local contentBottom = DialogFrame.Content.AbsolutePosition.Y + DialogFrame.Content.AbsoluteSize.Y
        local frameTop = DialogFrame.AbsolutePosition.Y
        local relativeY = contentBottom - frameTop
        ButtonContainer.Position = UDim2.new(0, 0, 0, relativeY + 20)

        local isClosing = false
        local function CloseDialog(instant)
            if isClosing then return end
            isClosing = true
            
            local fadeOutTime = instant and 0.2 or 0.4
            local fadeOutTween = TweenInfo.new(fadeOutTime, Enum.EasingStyle.Quint, Enum.EasingDirection.In)
            
            if not instant then
                for i = 1, 16 do
                    task.spawn(function()
                        local particle = SetProps(MakeElement("RoundFrame", DialogConfig.AccentColor, 0, 4), {
                            Size = UDim2.new(0, 6, 0, 6),
                            Position = UDim2.new(0.5, 0, 0.5, 0),
                            Parent = DialogFrame,
                            ZIndex = 10
                        })
                        
                        local angle = (i / 16) * math.pi * 2
                        local distance = math.random(60, 120)
                        local targetX = math.cos(angle) * distance
                        local targetY = math.sin(angle) * distance
                        
                        TweenService:Create(
                            particle,
                            TweenInfo.new(0.7, Enum.EasingStyle.Quart, Enum.EasingDirection.Out),
                            {
                                Position = UDim2.new(0.5, targetX, 0.5, targetY),
                                BackgroundTransparency = 1,
                                Size = UDim2.new(0, 0, 0, 0)
                            }
                        ):Play()
                    end)
                end
            end
            
            TweenService:Create(Overlay, fadeOutTween, {BackgroundTransparency = 1}):Play()
            TweenService:Create(DialogFrame, fadeOutTween, {
                BackgroundTransparency = 1,
                Size = UDim2.new(0, 300, 0, 0)
            }):Play()
            TweenService:Create(DialogStroke, fadeOutTween, {Transparency = 1}):Play()
            TweenService:Create(DialogFrame.Icon, fadeOutTween, {ImageTransparency = 1}):Play()
            TweenService:Create(DialogFrame.Title, fadeOutTween, {TextTransparency = 1}):Play()
            TweenService:Create(DialogFrame.Content, fadeOutTween, {TextTransparency = 1}):Play()
            
            for _, button in pairs(ButtonContainer:GetChildren()) do
                if button:IsA("TextButton") then
                    TweenService:Create(button, fadeOutTween, {BackgroundTransparency = 1}):Play()
                    if button:FindFirstChild("ButtonText") then
                        TweenService:Create(button.ButtonText, fadeOutTween, {TextTransparency = 1}):Play()
                    end
                    if button:FindFirstChildOfClass("UIStroke") then
                        TweenService:Create(button:FindFirstChildOfClass("UIStroke"), fadeOutTween, {Transparency = 1}):Play()
                    end
                end
            end
            
            task.wait(fadeOutTime)
            if Overlay and Overlay.Parent then
                Overlay:Destroy()
            end
        end

        -- ボタン作成
        for i, buttonData in ipairs(DialogConfig.Buttons) do
            local buttonWidth = (360 / #DialogConfig.Buttons) - 10
            local isPrimary = i == 1
            
            local Button = SetChildren(
                SetProps(MakeElement("Button"), {
                    Size = UDim2.new(0, buttonWidth, 0, 40),
                    BackgroundColor3 = isPrimary and DialogConfig.AccentColor or Color3.fromRGB(40, 40, 45),
                    BackgroundTransparency = 1,
                    LayoutOrder = i,
                    Parent = ButtonContainer
                }),
                {
                    MakeElement("Corner", 0, 8),
                    SetProps(MakeElement("Stroke", isPrimary and DialogConfig.AccentColor:Lerp(Color3.new(1,1,1), 0.3) or Color3.fromRGB(80, 80, 90), 1.5), {
                        Transparency = 1
                    }),
                    SetProps(MakeElement("Label", buttonData.Name, 15), {
                        Size = UDim2.new(1, 0, 1, 0),
                        Font = Enum.Font.GothamBold,
                        TextColor3 = isPrimary and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(200, 200, 210),
                        Name = "ButtonText",
                        TextTransparency = 1,
                        TextXAlignment = Enum.TextXAlignment.Center
                    })
                }
            )
            
            local ButtonStroke = Button:FindFirstChildOfClass("UIStroke")
            
            Button.MouseEnter:Connect(function()
                if isClosing then return end
                TweenService:Create(Button, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {
                    BackgroundColor3 = isPrimary and DialogConfig.AccentColor:Lerp(Color3.new(1,1,1), 0.2) or Color3.fromRGB(60, 60, 70)
                }):Play()
                TweenService:Create(ButtonStroke, TweenInfo.new(0.2), {Transparency = 0}):Play()
            end)
            
            Button.MouseLeave:Connect(function()
                if isClosing then return end
                TweenService:Create(Button, TweenInfo.new(0.2), {
                    BackgroundColor3 = isPrimary and DialogConfig.AccentColor or Color3.fromRGB(40, 40, 45)
                }):Play()
                TweenService:Create(ButtonStroke, TweenInfo.new(0.2), {Transparency = 0.5}):Play()
            end)
            
            Button.MouseButton1Click:Connect(function()
                if isClosing then return end
                pcall(buttonData.Callback)
                CloseDialog(false)
            end)
        end

        -- アニメーション
        task.wait()
        
        local fadeInTween = TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
        local quickFadeIn = TweenInfo.new(0.4, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
        
        TweenService:Create(Overlay, quickFadeIn, {BackgroundTransparency = 0.6}):Play()
        TweenService:Create(DialogFrame, fadeInTween, {BackgroundTransparency = 0}):Play()
        TweenService:Create(DialogStroke, quickFadeIn, {Transparency = 0.3}):Play()
        
        task.wait(0.1)
        TweenService:Create(DialogFrame.Icon, quickFadeIn, {ImageTransparency = 0}):Play()
        task.wait(0.05)
        TweenService:Create(DialogFrame.Title, quickFadeIn, {TextTransparency = 0}):Play()
        task.wait(0.05)
        TweenService:Create(DialogFrame.Content, quickFadeIn, {TextTransparency = 0}):Play()
        task.wait(0.1)
        
        for _, button in pairs(ButtonContainer:GetChildren()) do
            if button:IsA("TextButton") then
                TweenService:Create(button, quickFadeIn, {BackgroundTransparency = 0}):Play()
                if button:FindFirstChild("ButtonText") then
                    TweenService:Create(button.ButtonText, quickFadeIn, {TextTransparency = 0}):Play()
                end
                if button:FindFirstChildOfClass("UIStroke") then

                end

            if input.UserInputType == Enum.UserInputType.MouseButton1 then

                local framePos = DialogFrame.AbsolutePosition
                local frameSize = DialogFrame.AbsoluteSize
                
                if mousePos.X < framePos.X or mousePos.X > framePos.X + frameSize.X or
                   mousePos.Y < framePos.Y or mousePos.Y > framePos.Y + frameSize.Y then

                end
            end
        end)
    end)
end

function OrionLib:MakePopup(PopupConfig)
    task.spawn(function()
        PopupConfig = {

            Content = PopupConfig.Content or "This is a popup message",
            Image = PopupConfig.Image or "rbxassetid://4384403532",

            Position = PopupConfig.Position or "TopRight",

            AccentColor = PopupConfig.AccentColor or Color3.fromRGB(0, 200, 255)

        local PopupParent = SetProps(MakeElement("TFrame"), {
            Size = UDim2.new(0, 350, 0, 0),
            AutomaticSize = Enum.AutomaticSize.Y,
            Parent = ScreenGui,
            ClipsDescendants = false,
            BackgroundTransparency = 1,
            ZIndex = 500
        })

        local startPos, endPos
        if PopupConfig.Position == "TopRight" then
            startPos = UDim2.new(1, 400, 0, 20)
            endPos = UDim2.new(1, -370, 0, 20)
        elseif PopupConfig.Position == "TopLeft" then
            startPos = UDim2.new(0, -400, 0, 20)
            endPos = UDim2.new(0, 20, 0, 20)
        elseif PopupConfig.Position == "BottomRight" then
            startPos = UDim2.new(1, 400, 1, -120)
            endPos = UDim2.new(1, -370, 1, -120)
        elseif PopupConfig.Position == "BottomLeft" then
            startPos = UDim2.new(0, -400, 1, -120)
            endPos = UDim2.new(0, 20, 1, -120)
        else
            startPos = UDim2.new(0.5, -175, 0.5, -200)
            endPos = UDim2.new(0.5, -175, 0.5, -50)
        end

        PopupParent.Position = startPos

        local PopupFrame = SetChildren(
            SetProps(MakeElement("RoundFrame", PopupConfig.BackgroundColor, 0, 12), {
                Parent = PopupParent,
                Size = UDim2.new(1, 0, 0, 0),
                BackgroundTransparency = 1,
                AutomaticSize = Enum.AutomaticSize.Y
            }),
            {
                MakeElement("Padding", 15, 15, 15, 15),
                
                SetProps(MakeElement("Image", PopupConfig.Image), {
                    Size = UDim2.new(0, 28, 0, 28),
                    ImageColor3 = PopupConfig.AccentColor,
                    ImageTransparency = 1,
                    Name = "Icon"
                }),
                
                SetProps(MakeElement("Label", PopupConfig.Title, 16), {
                    Size = UDim2.new(1, -75, 0, 28),
                    Position = UDim2.new(0, 38, 0, 0),
                    Font = Enum.Font.GothamBold,
                    Name = "Title",
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextTransparency = 1
                }),
                
                SetProps(MakeElement("Label", PopupConfig.Content, 13), {
                    Size = UDim2.new(1, -10, 0, 0),
                    Position = UDim2.new(0, 0, 0, 36),
                    Font = Enum.Font.Gotham,
                    Name = "Content",
                    AutomaticSize = Enum.AutomaticSize.Y,
                    TextColor3 = Color3.fromRGB(200, 200, 200),
                    TextWrapped = true,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextTransparency = 1
                })
            }
        )

        local PopupStroke = SetProps(MakeElement("Stroke", PopupConfig.AccentColor, 1.5), {
            Transparency = 1,
            Parent = PopupFrame
        })

        local CloseBtn = SetChildren(
            SetProps(MakeElement("Button"), {
                Size = UDim2.new(0, 28, 0, 28),
                Position = UDim2.new(1, -34, 0, 0),
                BackgroundColor3 = Color3.fromRGB(40, 40, 45),
                BackgroundTransparency = 1,
                Parent = PopupFrame
            }),
            {
                MakeElement("Corner", 0, 14),
                SetProps(MakeElement("Stroke", Color3.fromRGB(80, 80, 90), 1.2), {
                    Transparency = 1
                }),
                SetProps(MakeElement("Image", "rbxassetid://7072725342"), {
                    Size = UDim2.new(0.5, 0, 0.5, 0),
                    Position = UDim2.new(0.25, 0, 0.25, 0),
                    ImageColor3 = Color3.fromRGB(200, 200, 210),
                    ImageTransparency = 1
                })
            }
        )

        local CloseIcon = CloseBtn:FindFirstChildOfClass("ImageLabel")
        local CloseStroke = CloseBtn:FindFirstChildOfClass("UIStroke")

        CloseBtn.MouseEnter:Connect(function()
            TweenService:Create(CloseBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(255, 60, 60), BackgroundTransparency = 0}):Play()
            TweenService:Create(CloseStroke, TweenInfo.new(0.2), {Color = Color3.fromRGB(255, 100, 100), Transparency = 0.3}):Play()
            TweenService:Create(CloseIcon, TweenInfo.new(0.2, Enum.EasingStyle.Back), {ImageTransparency = 0, Rotation = 90}):Play()
        end)

        CloseBtn.MouseLeave:Connect(function()
            TweenService:Create(CloseBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(40, 40, 45), BackgroundTransparency = 0.5}):Play()
            TweenService:Create(CloseStroke, TweenInfo.new(0.2), {Color = Color3.fromRGB(80, 80, 90), Transparency = 0.6}):Play()
            TweenService:Create(CloseIcon, TweenInfo.new(0.2), {ImageTransparency = 0.3, Rotation = 0}):Play()
        end)

        local isRemoving = false
        local function RemovePopup()
            if isRemoving then return end
            isRemoving = true
            
            local fadeOut = TweenInfo.new(0.3, Enum.EasingStyle.Quint, Enum.EasingDirection.In)
            
            TweenService:Create(PopupParent, fadeOut, {Position = startPos}):Play()
            TweenService:Create(PopupFrame, fadeOut, {BackgroundTransparency = 1}):Play()
            TweenService:Create(PopupStroke, fadeOut, {Transparency = 1}):Play()
            TweenService:Create(PopupFrame.Icon, fadeOut, {ImageTransparency = 1}):Play()
            TweenService:Create(PopupFrame.Title, fadeOut, {TextTransparency = 1}):Play()
            TweenService:Create(PopupFrame.Content, fadeOut, {TextTransparency = 1}):Play()
            TweenService:Create(CloseBtn, fadeOut, {BackgroundTransparency = 1}):Play()
            TweenService:Create(CloseIcon, fadeOut, {ImageTransparency = 1}):Play()
            TweenService:Create(CloseStroke, fadeOut, {Transparency = 1}):Play()
            
            task.wait(0.3)
            if PopupParent and PopupParent.Parent then
                PopupParent:Destroy()
            end
        end

        CloseBtn.MouseButton1Click:Connect(RemovePopup)

        task.wait()
        
        local slideIn = TweenInfo.new(0.6, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
        local fadeIn = TweenInfo.new(0.4, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
        
        TweenService:Create(PopupParent, slideIn, {Position = endPos}):Play()
        TweenService:Create(PopupFrame, fadeIn, {BackgroundTransparency = 0}):Play()
        TweenService:Create(PopupStroke, fadeIn, {Transparency = 0.5}):Play()
        
        task.wait(0.1)
        TweenService:Create(PopupFrame.Icon, fadeIn, {ImageTransparency = 0}):Play()
        task.wait(0.05)
        TweenService:Create(PopupFrame.Title, fadeIn, {TextTransparency = 0}):Play()
        task.wait(0.05)
        TweenService:Create(PopupFrame.Content, fadeIn, {TextTransparency = 0}):Play()
        task.wait(0.05)
        TweenService:Create(CloseBtn, fadeIn, {BackgroundTransparency = 0.5}):Play()
        TweenService:Create(CloseStroke, fadeIn, {Transparency = 0.6}):Play()
        TweenService:Create(CloseIcon, fadeIn, {ImageTransparency = 0.3}):Play()

        task.delay(PopupConfig.Time, RemovePopup)
    end)
end



function OrionLib:Init()
	if OrionLib.SaveCfg then	
		pcall(function()
			if isfile(OrionLib.Folder .. "/" .. game.GameId .. ".txt") then
				LoadCfg(readfile(OrionLib.Folder .. "/" .. game.GameId .. ".txt"))
				OrionLib:MakeNotification({
					Name = "設定",
					Content = "ゲーム " .. game.GameId .. " の設定を自動で読み込みました。",
					Time = 5
				})
			end
		end)		
	end	
end

    local function CreateParticles()
        if not ToggleConfig.EnableParticles then return end
        
        for i = 1, 8 do
            local Particle = SetProps(MakeElement("RoundFrame", ToggleConfig.Color, 0, 50), {
                Size = UDim2.new(0, 4, 0, 4),
                Position = UDim2.new(1, -24, 0.5, 0),
                AnchorPoint = Vector2.new(0.5, 0.5),
                Parent = ToggleFrame,
                ZIndex = 5
            })
            
            local angle = (i / 8) * math.pi * 2
            local distance = 30
            local targetX = math.cos(angle) * distance
            local targetY = math.sin(angle) * distance
            
            TweenService:Create(Particle, TweenInfo.new(0.6, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                Position = UDim2.new(1, -24 + targetX, 0.5, targetY),
                BackgroundTransparency = 1,
                Size = UDim2.new(0, 0, 0, 0)
            }):Play()
            
            task.delay(0.6, function()
                Particle:Destroy()
            end)
        end
    end

    -- グローエフェクト関数

        if not GlowFrame then return end

        if not RippleFrame then return end

    -- アニメーションスタイル別の処理
    local function GetAnimationInfo(style)
        if style == "Bounce" then
            return TweenInfo.new(ToggleConfig.AnimationSpeed, Enum.EasingStyle.Bounce, Enum.EasingDirection.Out)
        elseif style == "Elastic" then
            return TweenInfo.new(ToggleConfig.AnimationSpeed * 1.5, Enum.EasingStyle.Elastic, Enum.EasingDirection.Out)
        elseif style == "Pulse" then
            return TweenInfo.new(ToggleConfig.AnimationSpeed * 0.5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut)
        elseif style == "Flip" then
            return TweenInfo.new(ToggleConfig.AnimationSpeed, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
        else
            return TweenInfo.new(ToggleConfig.AnimationSpeed, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
        end
    end

    -- パルスアニメーション（有効時に繰り返し）
    local PulseConnection = nil
    local function StartPulseAnimation()
        if ToggleConfig.AnimationStyle ~= "Pulse" or not Toggle.Value then return end
        
        local function Pulse()
            if not Toggle.Value then return end
            
            TweenService:Create(ToggleBox, TweenInfo.new(0.8, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {
                Size = UDim2.new(0, 26, 0, 26)
            }):Play()
            
            task.wait(0.8)
            
            if Toggle.Value then
                TweenService:Create(ToggleBox, TweenInfo.new(0.8, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {
                    Size = UDim2.new(0, 24, 0, 24)
                }):Play()
            end
        end
        
        PulseConnection = task.spawn(function()
            while Toggle.Value and task.wait(1.6) do
                Pulse()
            end
        end)
    end

    -- 3Dフリップアニメーション
    local function PlayFlipAnimation()
        if ToggleConfig.AnimationStyle ~= "Flip" then return end
        
        -- 回転エフェクトをシミュレート
        TweenService:Create(ToggleBox, TweenInfo.new(ToggleConfig.AnimationSpeed * 0.5, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
            Size = UDim2.new(0, 4, 0, 24)
        }):Play()
        
        task.wait(ToggleConfig.AnimationSpeed * 0.5)
        
        TweenService:Create(ToggleBox, TweenInfo.new(ToggleConfig.AnimationSpeed * 0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, 24, 0, 24)
        }):Play()
    end

    function Toggle:Set(Value)
        Toggle.Value = Value
        
        local tweenInfo = GetAnimationInfo(ToggleConfig.AnimationStyle)
        
        -- フリップアニメーション
        if ToggleConfig.AnimationStyle == "Flip" then
            PlayFlipAnimation()
        end
        
        -- 基本アニメーション
        TweenService:Create(ToggleBox, tweenInfo, {
            BackgroundColor3 = Toggle.Value and ToggleConfig.Color or OrionLib.Themes.Default.Divider
        }):Play()
        
        TweenService:Create(ToggleBox.Stroke, tweenInfo, {
            Color = Toggle.Value and ToggleConfig.Color or OrionLib.Themes.Default.Stroke
        }):Play()
        
        -- アイコンアニメーション
        local iconSize = Toggle.Value and UDim2.new(0, 20, 0, 20) or UDim2.new(0, 8, 0, 8)
        if ToggleConfig.AnimationStyle == "Bounce" or ToggleConfig.AnimationStyle == "Elastic" then
            -- バウンス/エラスティックの場合は一旦大きくしてから戻す
            TweenService:Create(ToggleBox.Ico, TweenInfo.new(tweenInfo.Time * 0.6, tweenInfo.EasingStyle, Enum.EasingDirection.Out), {
                Size = Toggle.Value and UDim2.new(0, 24, 0, 24) or UDim2.new(0, 6, 0, 6)
            }):Play()
            
            task.wait(tweenInfo.Time * 0.6)
        end
        
        TweenService:Create(ToggleBox.Ico, tweenInfo, {
            ImageTransparency = Toggle.Value and 0 or 1,
            Size = iconSize
        }):Play()
        
        -- 追加エフェクト
        if Toggle.Value then
            CreateParticles()
            PlayGlowEffect(true)
        else
            PlayGlowEffect(false)
        end
        
        PlayRippleEffect()
        
        -- パルスアニメーション制御
        if Toggle.Value and ToggleConfig.AnimationStyle == "Pulse" then
            StartPulseAnimation()
        elseif PulseConnection then
            task.cancel(PulseConnection)
            PulseConnection = nil
        end
        
        ToggleConfig.Callback(Toggle.Value)
    end

					TweenService:Create(DropdownFrame.F.Ico,TweenInfo.new(.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),{Rotation = Dropdown.Toggled and 180 or 0}):Play()
					if #Dropdown.Options > MaxElements then

				Dropdown:Set(Dropdown.Value)

	local function UpdateSelectedDisplay()
		if #SelectedValues == 0 then
			if PlayerCountLabel then
				TweenService:Create(PlayerCountLabel, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0
				}):Play()
				PlayerCountLabel.Visible = true

				TextTransparency = 1
			}):Play()
			task.wait(DropdownConfig.AnimationSpeed)
			DropdownFrame.F.Selected.Visible = false
			DropdownFrame.F.Selected.Text = ""
		else
			if PlayerCountLabel then
				TweenService:Create(PlayerCountLabel, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 1
				}):Play()
				task.wait(DropdownConfig.AnimationSpeed)
				PlayerCountLabel.Visible = false
			end
			DropdownFrame.F.Selected.Visible = true
			
			local text = ""
			if Dropdown.MultipleSelection then
				for i, v in pairs(SelectedValues) do
					if i == 1 then
						text = v
					elseif i > 3 then
						text = text .. "..."
						break
					else
						text = text .. ", " .. v
					end
				end
			else
				text = SelectedValues[1] or ""
			end
			
			DropdownFrame.F.Selected.Text = text
			TweenService:Create(DropdownFrame.F.Selected, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				TextTransparency = 0
			}):Play()
		end
	end

			TweenService:Create(PlayerCountLabel, TweenInfo.new(DropdownConfig.AnimationSpeed / 2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				TextTransparency = 0.5

				TextTransparency = 0
			}):Play()

	end

	local searchDebounce = false
	local function FilterPlayers(query)
		if searchDebounce then return end
		searchDebounce = true
		
		for option, button in pairs(Dropdown.Buttons) do
			if query == "" then
				TweenService:Create(button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					BackgroundTransparency = button.BackgroundTransparency,
					Size = UDim2.new(1, 0, 0, 45)
				}):Play()
				TweenService:Create(button.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = button.Title.TextTransparency
				}):Play()
				TweenService:Create(button.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = button.Subtitle.TextTransparency
				}):Play()
				button.Visible = true
			else
				local match = string.find(string.lower(option), string.lower(query)) or 
							  string.find(string.lower(button.Subtitle.Text), string.lower(query))
				if match then
					button.Visible = true
					TweenService:Create(button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
						Size = UDim2.new(1, 0, 0, 45)
					}):Play()

					TweenService:Create(button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
						Size = UDim2.new(1, 0, 0, 0)
					}):Play()
					task.wait(DropdownConfig.AnimationSpeed)
					button.Visible = false
				end
			end
		end
		
		task.wait(0.05)
		searchDebounce = false
	end

	local function RemoveOption(Option)
		if Dropdown.Buttons[Option] then
			local button = Dropdown.Buttons[Option]
			
			TweenService:Create(button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				BackgroundTransparency = 1,
				Size = UDim2.new(1, 0, 0, 0)
			}):Play()
			TweenService:Create(button.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				TextTransparency = 1
			}):Play()
			TweenService:Create(button.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				TextTransparency = 1
			}):Play()
			if button:FindFirstChild("Icon") then
				TweenService:Create(button.Icon, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					ImageTransparency = 1
				}):Play()
			end
			
			task.wait(DropdownConfig.AnimationSpeed)
			button:Destroy()
			Dropdown.Buttons[Option] = nil
			Options = Options - 1
			UpdatePlayerCount()
			
			local n = table.find(SelectedValues, Option)
			if n then
				table.remove(SelectedValues, n)
				UpdateSelectedDisplay()
				DropdownConfig.Callback(Dropdown.MultipleSelection and SelectedValues or SelectedValues[1])
			end
		end
	end

	local function GetSortedPlayers()
		local players = PlayerService:GetPlayers()
		if DropdownConfig.SortPlayers then
			table.sort(players, function(a, b)
				return a.Name:lower() < b.Name:lower()
			end)
		end
		return players
	end

	for _, p in pairs(GetSortedPlayers()) do
		AddOption(p)
	end

	AddConnection(PlayerService.PlayerAdded, function(p) 
		AddOption(p)
	end)

	AddConnection(PlayerService.PlayerRemoving, function(p) 
		RemoveOption(p.Name)
	end)

	function Dropdown:Refresh() 
		for i, v in pairs(Dropdown.Buttons) do
			TweenService:Create(v, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				Size = UDim2.new(1, 0, 0, 0),
				BackgroundTransparency = 1
			}):Play()
			task.wait(0.02)
		end
		
		task.wait(DropdownConfig.AnimationSpeed)
		
		for i, v in pairs(Dropdown.Buttons) do
			v:Destroy()
			Dropdown.Buttons[i] = nil
		end
		Options = 0
		
		for _, p in pairs(GetSortedPlayers()) do
			AddOption(p)
			task.wait(0.02)
		end
		UpdatePlayerCount()
	end

	function Dropdown:Set(Value)
		local n = table.find(SelectedValues, Value)
		local Button = Dropdown.Buttons[Value]

		if not Button then return end
		
		if Dropdown.MultipleSelection then
			if not n then
				table.insert(SelectedValues, Value)
				TweenService:Create(Button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					BackgroundTransparency = 0
				}):Play()
				TweenService:Create(Button.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0
				}):Play()
				TweenService:Create(Button.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0
				}):Play()
			else
				table.remove(SelectedValues, n)
				TweenService:Create(Button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					BackgroundTransparency = 1
				}):Play()
				TweenService:Create(Button.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0.4
				}):Play()
				TweenService:Create(Button.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0.4
				}):Play()
			end
			
			Dropdown.Value = SelectedValues
			UpdateSelectedDisplay()
			return DropdownConfig.Callback(SelectedValues)
		else
			if n then
				table.remove(SelectedValues, n)
				TweenService:Create(Button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					BackgroundTransparency = 1
				}):Play()
				TweenService:Create(Button.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0.4
				}):Play()
				TweenService:Create(Button.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0.4
				}):Play()
				Dropdown.Value = nil
			else
				for _, v in pairs(Dropdown.Buttons) do
					TweenService:Create(v, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
						BackgroundTransparency = 1
					}):Play()
					TweenService:Create(v.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
						TextTransparency = 0.4
					}):Play()
					TweenService:Create(v.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
						TextTransparency = 0.4
					}):Play()
				end
				
				table.clear(SelectedValues)
				table.insert(SelectedValues, Value)

				TweenService:Create(Button, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					BackgroundTransparency = 0
				}):Play()
				TweenService:Create(Button.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0
				}):Play()
				TweenService:Create(Button.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
					TextTransparency = 0
				}):Play()
				
				Dropdown.Value = Value
			end
			
			UpdateSelectedDisplay()
			return DropdownConfig.Callback(Dropdown.Value)
		end
	end

	function Dropdown:Clear()
		table.clear(SelectedValues)
		Dropdown.Value = nil
		
		for _, v in pairs(Dropdown.Buttons) do
			TweenService:Create(v, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				BackgroundTransparency = 1
			}):Play()
			TweenService:Create(v.Title, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				TextTransparency = 0.4
			}):Play()
			TweenService:Create(v.Subtitle, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				TextTransparency = 0.4
			}):Play()
		end
		
		UpdateSelectedDisplay()
	end

	AddConnection(Click.MouseButton1Click, function()
		Dropdown.Toggled = not Dropdown.Toggled
		DropdownFrame.F.Line.Visible = Dropdown.Toggled
		
		TweenService:Create(DropdownFrame.F.Ico, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			Rotation = Dropdown.Toggled and 180 or 0
		}):Play()

		local extraHeight = DropdownConfig.SearchEnabled and 44 or 0

		if Options > MaxElements then
			TweenService:Create(DropdownFrame, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				Size = Dropdown.Toggled and UDim2.new(1, 0, 0, 38 + (MaxElements * 45) + extraHeight) or UDim2.new(1, 0, 0, 38)
			}):Play()
		else
			TweenService:Create(DropdownFrame, TweenInfo.new(DropdownConfig.AnimationSpeed, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				Size = Dropdown.Toggled and UDim2.new(1, 0, 0, 38 + (Options * 45) + extraHeight) or UDim2.new(1, 0, 0, 38)
			}):Play()
		end
	end)

	if DropdownConfig.Flag then				
		OrionLib.Flags[DropdownConfig.Flag] = Dropdown
	end
	
	UpdateSelectedDisplay()
	
	return Dropdown
end

							Holding = true
							BindConfig.Callback(Holding)
						else
							BindConfig.Callback()
						end

						local Key
						pcall(function()
							if not CheckKey(BlacklistedKeys, Input.KeyCode) then
								Key = Input.KeyCode
							end
						end)
						pcall(function()
							if CheckKey(WhitelistedMouse, Input.UserInputType) and not Key then
								Key = Input.UserInputType
							end
						end)

						Bind:Set(Key)
						SaveCfg(game.GameId)
